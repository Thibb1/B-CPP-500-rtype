\DoxyCodeLine{81 \textcolor{comment}{-\/-\/ Data Format}}
\DoxyCodeLine{82 \textcolor{comment}{}}
\DoxyCodeLine{83 \textcolor{comment}{A QOI file has a 14 byte header, followed by any number of data "{}chunks"{} and an}}
\DoxyCodeLine{84 \textcolor{comment}{8-\/byte end marker.}}
\DoxyCodeLine{85 \textcolor{comment}{}}
\DoxyCodeLine{86 \textcolor{comment}{struct qoi\_header\_t \{}}
\DoxyCodeLine{87 \textcolor{comment}{    char     magic[4];   // magic bytes "{}qoif"{}}}
\DoxyCodeLine{88 \textcolor{comment}{    uint32\_t width;      // image width in pixels (BE)}}
\DoxyCodeLine{89 \textcolor{comment}{    uint32\_t height;     // image height in pixels (BE)}}
\DoxyCodeLine{90 \textcolor{comment}{    uint8\_t  channels;   // 3 = RGB, 4 = RGBA}}
\DoxyCodeLine{91 \textcolor{comment}{    uint8\_t  colorspace; // 0 = sRGB with linear alpha, 1 = all channels linear}}
\DoxyCodeLine{92 \textcolor{comment}{\};}}
\DoxyCodeLine{93 \textcolor{comment}{}}
\DoxyCodeLine{94 \textcolor{comment}{Images are encoded row by row, left to right, top to bottom. The decoder and}}
\DoxyCodeLine{95 \textcolor{comment}{encoder start with \{r: 0, g: 0, b: 0, a: 255\} as the previous pixel value. An}}
\DoxyCodeLine{96 \textcolor{comment}{image is complete when all pixels specified by width * height have been covered.}}
\DoxyCodeLine{97 \textcolor{comment}{}}
\DoxyCodeLine{98 \textcolor{comment}{Pixels are encoded as}}
\DoxyCodeLine{99 \textcolor{comment}{ -\/ a run of the previous pixel}}
\DoxyCodeLine{100 \textcolor{comment}{ -\/ an index into an array of previously seen pixels}}
\DoxyCodeLine{101 \textcolor{comment}{ -\/ a difference to the previous pixel value in r,g,b}}
\DoxyCodeLine{102 \textcolor{comment}{ -\/ full r,g,b or r,g,b,a values}}
\DoxyCodeLine{103 \textcolor{comment}{}}
\DoxyCodeLine{104 \textcolor{comment}{The color channels are assumed to not be premultiplied with the alpha channel}}
\DoxyCodeLine{105 \textcolor{comment}{("{}un-\/premultiplied alpha"{}).}}
\DoxyCodeLine{106 \textcolor{comment}{}}
\DoxyCodeLine{107 \textcolor{comment}{A running array[64] (zero-\/initialized) of previously seen pixel values is}}
\DoxyCodeLine{108 \textcolor{comment}{maintained by the encoder and decoder. Each pixel that is seen by the encoder}}
\DoxyCodeLine{109 \textcolor{comment}{and decoder is put into this array at the position formed by a hash function of}}
\DoxyCodeLine{110 \textcolor{comment}{the color value. In the encoder, if the pixel value at the index matches the}}
\DoxyCodeLine{111 \textcolor{comment}{current pixel, this index position is written to the stream as QOI\_OP\_INDEX.}}
\DoxyCodeLine{112 \textcolor{comment}{The hash function for the index is:}}
\DoxyCodeLine{113 \textcolor{comment}{}}
\DoxyCodeLine{114 \textcolor{comment}{    index\_position = (r * 3 + g * 5 + b * 7 + a * 11) \% 64}}
\DoxyCodeLine{115 \textcolor{comment}{}}
\DoxyCodeLine{116 \textcolor{comment}{Each chunk starts with a 2-\/ or 8-\/bit tag, followed by a number of data bits. The}}
\DoxyCodeLine{117 \textcolor{comment}{bit length of chunks is divisible by 8 -\/ i.e. all chunks are byte aligned. All}}
\DoxyCodeLine{118 \textcolor{comment}{values encoded in these data bits have the most significant bit on the left.}}
\DoxyCodeLine{119 \textcolor{comment}{}}
\DoxyCodeLine{120 \textcolor{comment}{The 8-\/bit tags have precedence over the 2-\/bit tags. A decoder must check for the}}
\DoxyCodeLine{121 \textcolor{comment}{presence of an 8-\/bit tag first.}}
\DoxyCodeLine{122 \textcolor{comment}{}}
\DoxyCodeLine{123 \textcolor{comment}{The byte stream's end is marked with 7 0x00 bytes followed a single 0x01 byte.}}
\DoxyCodeLine{124 \textcolor{comment}{}}
\DoxyCodeLine{125 \textcolor{comment}{}}
\DoxyCodeLine{126 \textcolor{comment}{The possible chunks are:}}
\DoxyCodeLine{127 \textcolor{comment}{}}
\DoxyCodeLine{128 \textcolor{comment}{}}
\DoxyCodeLine{129 \textcolor{comment}{.-\/ QOI\_OP\_INDEX -\/-\/-\/-\/-\/-\/-\/-\/-\/-\/.}}
\DoxyCodeLine{130 \textcolor{comment}{|         Byte[0]         |}}
\DoxyCodeLine{131 \textcolor{comment}{|  7  6  5  4  3  2  1  0 |}}
\DoxyCodeLine{132 \textcolor{comment}{|-\/-\/-\/-\/-\/-\/-\/+-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/|}}
\DoxyCodeLine{133 \textcolor{comment}{|  0  0 |     index       |}}
\DoxyCodeLine{134 \textcolor{comment}{`-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/`}}
\DoxyCodeLine{135 \textcolor{comment}{2-\/bit tag b00}}
\DoxyCodeLine{136 \textcolor{comment}{6-\/bit index into the color index array: 0..63}}
\DoxyCodeLine{137 \textcolor{comment}{}}
\DoxyCodeLine{138 \textcolor{comment}{A valid encoder must not issue 2 or more consecutive QOI\_OP\_INDEX chunks to the}}
\DoxyCodeLine{139 \textcolor{comment}{same index. QOI\_OP\_RUN should be used instead.}}
\DoxyCodeLine{140 \textcolor{comment}{}}
\DoxyCodeLine{141 \textcolor{comment}{}}
\DoxyCodeLine{142 \textcolor{comment}{.-\/ QOI\_OP\_DIFF -\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/.}}

\DoxyCodeLine{146 \textcolor{comment}{|  0  1 |  dr |  dg |  db |}}
\DoxyCodeLine{147 \textcolor{comment}{`-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/`}}
\DoxyCodeLine{148 \textcolor{comment}{2-\/bit tag b01}}
\DoxyCodeLine{149 \textcolor{comment}{2-\/bit   red channel difference from the previous pixel between -\/2..1}}
\DoxyCodeLine{150 \textcolor{comment}{2-\/bit green channel difference from the previous pixel between -\/2..1}}

\DoxyCodeLine{152 \textcolor{comment}{}}
\DoxyCodeLine{153 \textcolor{comment}{The difference to the current channel values are using a wraparound operation,}}

\DoxyCodeLine{155 \textcolor{comment}{}}

\DoxyCodeLine{157 \textcolor{comment}{0 (b00). 1 is stored as 3 (b11).}}
\DoxyCodeLine{158 \textcolor{comment}{}}

\DoxyCodeLine{167 \textcolor{comment}{`-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/-\/`}}

\DoxyCodeLine{177 \textcolor{comment}{    db\_dg = (cur\_px.b -\/ prev\_px.b) -\/ (cur\_px.g -\/ prev\_px.g)}}
