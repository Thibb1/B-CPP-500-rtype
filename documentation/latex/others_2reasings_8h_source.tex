\DoxyCodeLine{11 \textcolor{comment}{*   The four inputs t,b,c,d are defined as follows:}}

\DoxyCodeLine{15 \textcolor{comment}{*   d = total time it should take to complete (duration)}}
\DoxyCodeLine{16 \textcolor{comment}{*}}
\DoxyCodeLine{17 \textcolor{comment}{*   Example:}}
\DoxyCodeLine{18 \textcolor{comment}{*}}
\DoxyCodeLine{19 \textcolor{comment}{*   int currentTime = 0;}}

\DoxyCodeLine{22 \textcolor{comment}{*   float finalPositionX = 30.0f;}}
\DoxyCodeLine{23 \textcolor{comment}{*   float currentPositionX = startPositionX;}}
\DoxyCodeLine{24 \textcolor{comment}{*}}
